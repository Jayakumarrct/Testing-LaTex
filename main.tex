\begin{thmbox}
  \begin{theorem}
    The identity element of a group is unique.
  \end{theorem}
  \begin{proof}
    If $e$ and $e'$ are identities, then $e=e\cdot e'=e'$.
  \end{proof}
\end{thmbox}
